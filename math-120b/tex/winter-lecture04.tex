\lecture{}

\begin{definition}
Let $E/F$ be an algebraic extension. The \define{Normal Closure}
$\normalclosure{E}$ of $F$ is the intersection of all $K$ such that
$\closure{F}/K/F$ and $K/F$ is a normal extension.
\end{definition}

\begin{definition}
Let $E/F$ be an (algebraic?) extension.
Let $\sigma\in\hom_{F}(E,\closure{F})\setminus\{0\}$ be an $F$-embedding.
We call $\sigma(E)\subset\closure{F}$ a \define{Conjugate} of $E$ in $\closure{F}$.
\end{definition}

\begin{definition}
Let $E/F$ be any field extension.
For an algebraic element $\alpha\in E$ over $F$ and for any distinct
$F$-embeddings $\sigma_{1}$, \dots, $\sigma_{n}\in\hom_{F}(F(\alpha),\closure{F})$,
we call the elements $\sigma_{1}(\alpha)$, \dots, $\sigma_{n}(\alpha)$
\define{Conjugates} of $\alpha$ (= the set of all distinct roots of
the minimal polynomial $p_{\alpha}(x)\in F[x]$), and
$F(\sigma_{1}(\alpha),\dots,\sigma_{n}(\alpha))$ is the smallest
normal extension of $F$ containing some $\sigma_{i}(\alpha)$.
\end{definition}

\begin{proposition}
The normal closure $\normalclosure{E}$ is generrated by all conjugates
of $E$ in $\closure{F}$.
\end{proposition}

\begin{proof}
By Theorem~\ref{thm:well-definedness:normal-extension}~\ref{criteria:normal:1},
all the images of $E$ is contained in $\normalclosure{E}$ so
$\normalclosure{E}$ contains all the conjugates of $E$ in $\closure{F}$.
On the other hand, the field generated by all conjugates of $E$
satisfies Theorem~\ref{thm:well-definedness:normal-extension}~\ref{criteria:normal:1}
by Definition, so it contains $\normalclosure{E}$.
\end{proof}

\subsection{Separable extensions}

\begin{definition}
Let $E/F$ be an algebraic extension. The \define{Separable Degree}
$[E:F]_{s}:=|\hom_{F}(E,\closure{F})|$.
\end{definition}

\begin{proposition}\label{prop:4.1}
Let $L/E$ and $E/F$ be algebraic extensions.
Then $[L:F]_{s}=[L:E]_{s}[E:F]_{s}$.

Moreover, $[L:F]_{s}\leq[L:F]$. Further $[L:F]_{s}=[L:F]$ if and only
if $[L:E]_{s}=[L:E]$ and $[E:F]_{s}=[E:F]$.
\end{proposition}

\begin{proof}
\textsc{Claim 1:} separable degree is multiplicative.
By Proposition~\ref{prop:2.8}, for any $f\in\hom_{F}(E,\closure{F})$
is extended to $\overline{f}\in\Aut_{F}(\closure{F}$. To show the map
\begin{equation}
\begin{split}
\Phi\colon&\hom_{F}(E,\closure{F})\times\hom(L,\closure{F})\to\hom_{F}(L,\closure{F})\\
&(f,g)\mapsto\overline{f}\circ g
\end{split}
\end{equation}
is bijective, let us construct its inverse. For $h\in\hom_{F}(L,\closure{F})$,
the restriction $h|_{E}\in\hom_{F}(E,\closure{F})$ is extended to
$\overline{h|_{E}}\in\Aut_{F}(\closure{F})$ by Proposition~\ref{prop:2.8}.
Then $(\overline{h|_{E}})^{-1}\circ h\in\hom_{F}(L,\closure{F})$ is an
$E$-morphism
\begin{equation}
(\overline{h|_{E}})^{-1}\circ h|_{E}=(h^{-1}\circ h)|_{E}=\id_{E}.
\end{equation}
Then the map
\begin{equation}
\begin{split}
\Psi\colon&\hom_{F}(L,\closure{F})\to\hom_{F}(E,\closure{F})\times\hom(L,\closure{F})\\
&h\mapsto(h|_{E},(\overline{h|_{E}})^{-1}\circ h)
\end{split}
\end{equation}
is the inverse of $\Phi$. In fact,
\begin{equation}
\Phi\circ\Psi(h)=\Phi(h|_{E},(\overline{h|_{E}})^{-1}\circ h)=\overline{h|_{E}}\circ(\overline{h|_{E}})^{-1}\circ h=h,
\end{equation}
and
\begin{subequations}
  \begin{align}
\Psi\circ\Phi(f,g)
&=\Psi(\overline{f}\circ g)\\
&=((\overline{f}\circ g)|_{E},(\overline{\overline{f}\circ g})^{-1}\circ(\overline{f}\circ g))\\
&=(\overline{f|_{E}},\overline{f}^{-1}\circ\overline{f}\circ g)\\
&=(f,g).
  \end{align}
\end{subequations}
Hence the first claim.

\textsc{Claim 2:} separable degree of a field extension is bounded by the degree
of the extension. Without loss of generality, assume $E/F$ is a finite
extension. We see that we always have
\begin{equation}
[E:F]_{s}\leq[E:F],
\end{equation}
and for any intermediate field subextension $E/K/F$ we claim
$[E:F]_{s}=[E:F]$ if and only if $[E:K]_{s}=[E:K]$ and $[K:F]_{s}=[K:F]$.
So there exists $\alpha_{1},\dots,\alpha_{n}\in E$ such that
\begin{equation}
E = F(\alpha_{1},\dots,\alpha_{n})
\end{equation}
and there is some $m\leq n$ such that
\begin{equation}
K = F(\alpha_{1},\dots,\alpha_{m}).
\end{equation}
Writing a tower of simple field extensions $F_{0}=F$ and
$F_{i+1}=F_{i}(\alpha_{i+1})$. Then
\begin{equation}
E=F_{n}/F_{n-1}/\dots/F_{m}/\dots/F_{1}/F_{0}=F.
\end{equation}
By Proposition~\ref{prop:2.5}, letting $p_{\alpha_{i+1}}\in F_{i}[x]$ be
the minimal polynomial of $\alpha_{i+1}$, 
\begin{equation}
[F_{i+1}:F_{i}]_{s}\leq\deg(p_{\alpha_{i}})=[F_{i+1}:F_{i}].
\end{equation}
By the preceding discussion,
\begin{subequations}
  \begin{align}
[E:F]_{s} &= \prod^{n}_{i=1}[F_{i+1}:F_{i}]_{s}\\
&\leq\prod^{n}_{i=1}[F_{i+1}:F_{i}]=[E:F].
  \end{align}
\end{subequations}
Hence the result.
\end{proof}

\begin{definition}
Let $E/F$ be a finite extension. We call it a \define{Separable Extension}
if $[E:F]_{s}=[E:F]$.
\end{definition}

\begin{remark}
An algebraic extension $E/F$ is separable if all its finitely
generated subextensions are separable.
\end{remark}

\begin{definition}
Let $E/F$ be a field extension, let $\alpha\in E$ be algebraic over $F$.
We say $\alpha$ is \define{Separable} if $F(\alpha)/F$ is separable.

By Proposition~\ref{prop:2.5}, this is equivalent to saying the
minimal polynomial $p_{\alpha}\in F[x]$ is separable (i.e., has no
multiple roots).
\end{definition}

\begin{node}
If $\alpha$ is a root of a separaable polynomial $g(x)\in F[x]$,
then the minimal polynomial $p_{\alpha}\in F[x]$ divides $g(x)$ (so
$\alpha$ is separable).
\end{node}

\begin{node}
Let $L/E$ and $E/F$ be field extensions. If $\alpha\in L$ is separable
over $F$, then $\alpha$ is separable over $E$.
\end{node}

\begin{corollary}\label{cor:4.2}
Let $E/F$ be an algebraic extension. Then
$E/F$ is separable if and only if every $\alpha\in E$ is
separable. (Equivalently, if every generator of $E$ is separable.)
\end{corollary}

\begin{proof}
\textsc{Case 1:} the finite case (which follows from Proposition~\ref{prop:4.1}).

\forwardproof\ We have $E/F(\alpha)/F$ and
\begin{equation}
[E:F]_{s} = [E:F(\alpha)]_{s}[F(\alpha):F]_{s}
\end{equation}
and
\begin{equation}
[E:F]_{s} = [E:F] = [E:F(\alpha)][F(\alpha):F],
\end{equation}
hence $F(\alpha)/F$ is separable, hence $\alpha$ is separable.

\backwardproof\ Setting $E=F_{n}/\dots/F_{0}=F$ where
$F_{i+1}=F_{i}(\alpha_{i+1})$ and $\alpha_{1}$, \dots, $\alpha_{n}$
generate $E$, then at each step $\alpha_{i+1}$ is separable over
$F_{i}$ (hence all the $\alpha_{i}$ are separable over $F$).

\textsc{Case 2:} the general case.

\forwardproof\ By definition, since every finite subextension is
sepraable.

\backwardproof\ Every finite subfield extension $F'$ lies in some
$F(\alpha_{1},\dots,\alpha_{n})\subset E$, which is separable (by the
first case). Hence every element in $F'$ are separable. Hence $E/F$ is
separable.  
\end{proof}
