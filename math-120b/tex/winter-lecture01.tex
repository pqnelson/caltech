\begin{proposition}\label{prop:1.2}
If $L_{1}/L_{2}$ is a simple extension $L_{1}=L_{2}(\alpha)$, then
\begin{equation}
[L_{1}:L_{2}]=\deg(p_{\alpha}),
\end{equation}
where $p_{\alpha}\in L_{1}[x]$ is the
minimal polynomial of $\alpha$.
\end{proposition}
