\lecture{}

Today was just reviewing the midterm's more difficult problems,
because not many people attended last Friday's lecture.

\begin{xca}
Is $\CC[x,y]/(x^{2},xy^{2})$ Artinian or not?
\end{xca}

\begin{node}
One way to solve this is to look at
$(x,y)\supset(x^{2},xy^{2},y^{n})\supset(x^{2},xy^{2})$ and realize we
can keep adding more ideals.

Another way is to use the fact that
\begin{equation*}
\mbox{Artinian} = \mbox{Noetherian} + \mbox{Finite-Dimensional}
\end{equation*}
The ring corresponds to solutions of the system of equations
\begin{equation}
\left.\begin{array}{rcc}
  x^{2} & = & 0\\
  xy^{2} & = & 0
\end{array}\right\}\implies x=0, y\in\CC\mbox{ arbitrary}
\end{equation}
This is 1-dimensional.
\end{node}

\begin{xca}
Consider the map $\phi\colon\Spec(\ZZ[x]/(2x,x^{2}-x))\to\Spec(\ZZ)$
corresponding to the canonical $\ZZ\to\ZZ[x]/(2x,x^{2}-x)$. Does
going-up hold for $\phi$? Does going-down hold for $\phi$?
\end{xca}

\begin{node}
To answer the previous exercise, we need to first compute $\Spec(\ZZ[x]/(2x,x^{2}-x))$.
We see
\begin{subequations}
  \begin{align}
    (x+1)(x+2)&=x^{2}+3x+2\\
    &\equiv x+3x+2\bmod{(2x,x^{2}-x)}\\
    &\equiv2\bmod{(2x,x^{2}-x)}.
  \end{align}
\end{subequations}
Then the possible prime ideals are $(2,x)$, $(x)$, $(x+1)$, and the
principal ideals $(p)$ each generated by an odd prime integer $p=3,5,7,11,\dots$.

We see $(2)\subset(x+1)$. So $(x+1)$ is lying over $(2)$, but there's
nothing in $\Spec(\ZZ[x]/(2x,x^{2}-x))$ lying over the zero ideal of $\ZZ$.
\end{node}

\begin{xca}
Let $A=\CC[x,y]$ be our ring, let $\mathfrak{m}=(x,y)$ be a module
over $A$.
\begin{enumerate}
\item Is $\mathfrak{m}/\mathfrak{m}^{2}$ flat? Is it projective?
\item What is $\dim_{A}(\mathfrak{m}/\mathfrak{m}^{2})$?
\item Let $L=\CC(x,y)$. What is $\dim_{L}(\mathfrak{m}\otimes_{A}L)$?
\end{enumerate}
\end{xca}
